METHOD_BODY::= STATEMENT_LIST
STATEMENT_LIST ::= STATEMENT | STATEMENT STATEMENT_LIST_A
STATEMENT_LIST_A ::= STATEMENT STATEMENT_LIST_A | \L
STATEMENT ::= DECLARATION
| IF
| WHILE
| ASSIGNMENT
DECLARATION ::= PRIMITIVE_TYPE IDENTIFIER ;
PRIMITIVE_TYPE ::= int | float
IF ::= if ( EXPRESSION ) { STATEMENT} else { STATEMENT }
WHILE ::= while ( EXPRESSION ) { STATEMENT }
ASSIGNMENT ::= IDENTIFIER = EXPRESSION ;
EXPRESSION ::= NUMBER
| INFIX_OPERATOR EXPRESSION EXPRESSION_A
| IDENTIFIER
| ( EXPRESSION )
INFIX_OPERATOR ::= + | - | * | / | %| < | > | <= | >= | == | != | \| | &&
EXPRESSION_A ::= INFIX_OPERATOR EXPRESSION EXPRESSION_A | \L
